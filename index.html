jQuery slider demo

Установка

    <link href="path/to/slider.css" rel="stylesheet">  <!-- Стили слайдера -->
    <script src="path/to/jquery.js"></script>          <!-- jQuery версии 1.8 или выше -->
    <script src="path/to/fullsceen.js"></script>       <!-- [Необязательно] Библиотека для поддержки полноэкранного режима -->
    <script src="path/to/slider.js"></script>          <!-- Основной файл -->

Использование

    $('#selector').slider(options);

Доступные настройки и их значения по-умолчанию

    $('#selector').slider({
      items: '.slide',                  // Селектор элементов-слайдов
      firstSlide: 0,                    // Индекс первого слайда
      width: 400,                       // Ширина слайдера при отображении на странице (px)
      originalWidth: 1000,              // Ширина слайда (px)
      proportions: 4/3,                 // Пропорции слайдера
      backgroundFullScreen: '#222',     // Фон полноэкранного слайдера
      background: '#fff',               // Фон слайда
      hideControls: false,              // Скрыть элементы управления
      ajax: [],                         // Ссылки на слайды, которые должны загружаться Ajax'ом
      ajaxCache: true,                  // Кэшировать ли Ajax ответы
                                        // Переход между слайдами
      effect: function(oldSlide, newSlide) {
        if (oldSlide) {
          oldSlide.fadeOut(200);
        }
        if (newSlide) {
          newSlide.fadeIn(200);
        }
      }
    });

Элементы управления

      var slider3 = $('#slider-3').slider({
        hideControls: true
      });

      $('.prev').click(function() {
        var current = slider3.goPrev();
        $('.current').text(current);
      });

      $('.next').click(function() {
        var current = slider3.goNext();
        $('.current').text(current);
      });

      $('.fs').click(function() {
        slider3.fullScreen();
      });

Эффект смены слайда

      $('#slider-5').slider({
        effect: function(oldSlide, newSlide, index) {
          if (oldSlide) {
            index % 2 ? oldSlide.slideUp(200) : oldSlide.hide(200);
          }
          if (newSlide) {
            index % 2 ? newSlide.slideDown(200) : newSlide.show(200);
          }
        }
      });
